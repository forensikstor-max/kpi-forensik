import React, { useState, useEffect, useMemo } from 'react';
import { 
  Plus, 
  Search, 
  Trash2, 
  Edit2, 
  Printer, 
  Save, 
  X,
  Book,
  Clock,
  Calendar,
  CheckCircle2,
  AlertCircle,
  FilterX,
  PieChart,
  Activity,
  BarChart3,
  Filter,
  AlertTriangle
} from 'lucide-react';

const App = () => {
  // State untuk data utama
  const [records, setRecords] = useState([]);
  const [searchTerm, setSearchTerm] = useState('');
  const [selectedMonth, setSelectedMonth] = useState('Semua');
  const [isEditing, setIsEditing] = useState(null);
  const [showError, setShowError] = useState(false);
  
  const senaraiBulan = [
    "Januari", "Februari", "Mac", "April", "Mei", "Jun", 
    "Julai", "Ogos", "September", "Oktober", "November", "Disember"
  ];

  // State untuk borang
  const [formData, setFormData] = useState({
    rn_buku: '',
    exclusion: 'Tidak',
    waktu_terima: '',
    waktu_serah: '',
    jumlah_jam: 0,
    status_kpi: ''
  });

  // Load data dari localStorage
  useEffect(() => {
    const savedData = localStorage.getItem('kpi_kematian_records_v3');
    if (savedData) {
      setRecords(JSON.parse(savedData));
    }
  }, []);

  // Simpan data ke localStorage
  useEffect(() => {
    localStorage.setItem('kpi_kematian_records_v3', JSON.stringify(records));
  }, [records]);

  // Fungsi untuk mendapatkan bulan daripada string tarikh
  const getMonthNameFromDate = (dateStr) => {
    if (!dateStr) return "";
    const date = new Date(dateStr);
    return senaraiBulan[date.getMonth()];
  };

  // Filter rekod mengikut bulan terlebih dahulu
  const recordsByMonth = useMemo(() => {
    if (selectedMonth === 'Semua') return records;
    return records.filter(r => {
      const dateToCompare = r.waktu_terima || new Date(r.id).toISOString();
      return getMonthNameFromDate(dateToCompare) === selectedMonth;
    });
  }, [records, selectedMonth]);

  // Pengiraan Statistik berdasarkan rekod yang telah difilter mengikut bulan
  const stats = useMemo(() => {
    const total = recordsByMonth.length;
    const tercapai = recordsByMonth.filter(r => r.status_kpi === 'Tercapai').length;
    const tidakTercapai = recordsByMonth.filter(r => r.status_kpi === 'Tidak Tercapai').length;
    const exclusion = recordsByMonth.filter(r => r.status_kpi === 'Exclusion').length;
    
    const denominator = total - exclusion;
    const numerator = tercapai;
    const peratus = denominator > 0 ? ((numerator / denominator) * 100).toFixed(2) : "0.00";

    return { total, tercapai, tidakTercapai, exclusion, denominator, numerator, peratus };
  }, [recordsByMonth]);

  // Fungsi mengira beza jam dan status KPI
  const calculateKPI = (terima, serah, exclusion) => {
    if (exclusion === 'Ya') return { jam: 0, kpi: 'Exclusion' };
    if (!terima || !serah) return { jam: 0, kpi: '' };
    
    const start = new Date(terima);
    const end = new Date(serah);
    const diffMs = end.getTime() - start.getTime();
    
    if (diffMs < 0) return { jam: 0, kpi: 'Ralat Masa' };
    
    const diffHrs = (diffMs / (1000 * 60 * 60)).toFixed(2);
    const isMet = parseFloat(diffHrs) < 3;
    
    return {
      jam: diffHrs,
      kpi: isMet ? 'Tercapai' : 'Tidak Tercapai'
    };
  };

  const handleInputChange = (e) => {
    const { name, value } = e.target;
    let newFormData = { ...formData, [name]: value };
    
    // Reset masa jika exclusion Ya
    if (name === 'exclusion' && value === 'Ya') {
      newFormData.waktu_terima = '';
      newFormData.waktu_serah = '';
    }

    const { jam, kpi } = calculateKPI(
      newFormData.waktu_terima,
      newFormData.waktu_serah,
      newFormData.exclusion
    );
    
    newFormData.jumlah_jam = jam;
    newFormData.status_kpi = kpi;
    
    setShowError(kpi === 'Ralat Masa');
    setFormData(newFormData);
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    
    if (formData.status_kpi === 'Ralat Masa') {
      setShowError(true);
      return;
    }
    
    if (isEditing) {
      setRecords(records.map(rec => rec.id === isEditing ? { ...formData, id: isEditing } : rec));
      setIsEditing(null);
    } else {
      const newRecord = {
        ...formData,
        id: Date.now(),
      };
      setRecords([newRecord, ...records]);
    }

    setFormData({ rn_buku: '', exclusion: 'Tidak', waktu_terima: '', waktu_serah: '', jumlah_jam: 0, status_kpi: '' });
    setShowError(false);
  };

  const deleteRecord = (id) => {
    // Penggantian alert dengan confirmation UI mudah
    if (window.confirm('Adakah anda pasti ingin memadam rekod ini?')) {
      setRecords(records.filter(rec => rec.id !== id));
    }
  };

  const startEdit = (record) => {
    setIsEditing(record.id);
    setFormData(record);
    setShowError(record.status_kpi === 'Ralat Masa');
    window.scrollTo({ top: 0, behavior: 'smooth' });
  };

  const cancelEdit = () => {
    setIsEditing(null);
    setFormData({ rn_buku: '', exclusion: 'Tidak', waktu_terima: '', waktu_serah: '', jumlah_jam: 0, status_kpi: '' });
    setShowError(false);
  };

  const handlePrint = () => {
    window.print();
  };

  const filteredRecords = recordsByMonth.filter(rec => 
    rec.rn_buku.toLowerCase().includes(searchTerm.toLowerCase()) ||
    rec.status_kpi.toLowerCase().includes(searchTerm.toLowerCase())
  );

  return (
    <div className="min-h-screen bg-slate-50 p-4 md:p-8 font-sans text-slate-800">
      <div className="max-w-7xl mx-auto">
        {/* Header Seksyen */}
        <header className="mb-8 flex flex-col md:flex-row md:items-start justify-between gap-4 border-b-2 border-slate-200 pb-6">
          <div className="flex gap-4 items-start">
            <div className="bg-blue-600 p-3 rounded-2xl text-white shadow-lg print:bg-blue-600">
              <Book size={32} />
            </div>
            <div>
              <h1 className="text-2xl md:text-3xl font-black text-slate-900 tracking-tight leading-none uppercase">
                Data Pendaftaran Kematian
              </h1>
              <h2 className="text-lg md:text-xl font-bold text-slate-700 mt-1 uppercase tracking-wide">
                Unit Perubatan Forensik
              </h2>
              <p className="text-slate-500 font-semibold tracking-wider">
                Hospital Tengku Permaisuri Norashikin
              </p>
              <div className="mt-2 inline-flex items-center gap-2 bg-blue-50 px-3 py-1 rounded-full border border-blue-100 print:bg-white print:border-slate-200">
                <Calendar size={14} className="text-blue-600" />
                <span className="text-xs font-bold text-blue-700 uppercase tracking-tighter">
                  Laporan: {selectedMonth === 'Semua' ? 'Keseluruhan Tahun' : `Bulan ${selectedMonth}`}
                </span>
              </div>
            </div>
          </div>
          
          <div className="flex items-center gap-3 print:hidden self-end md:self-start">
            <div className="relative">
              <Filter className="absolute left-3 top-1/2 -translate-y-1/2 text-slate-400" size={16} />
              <select 
                value={selectedMonth}
                onChange={(e) => setSelectedMonth(e.target.value)}
                className="pl-9 pr-4 py-2.5 bg-white border border-slate-200 rounded-xl shadow-sm focus:ring-2 focus:ring-blue-500 outline-none transition-all font-semibold text-slate-700"
              >
                <option value="Semua">Semua Bulan</option>
                {senaraiBulan.map(bulan => (
                  <option key={bulan} value={bulan}>{bulan}</option>
                ))}
              </select>
            </div>
            
            <button 
              onClick={handlePrint}
              className="flex items-center justify-center gap-2 bg-slate-900 hover:bg-black text-white px-6 py-2.5 rounded-xl transition-all shadow-md active:scale-95"
            >
              <Printer size={20} />
              Cetak
            </button>
          </div>
        </header>

        {/* Paparan Ringkas (Dashboard Stats) */}
        <div className="grid grid-cols-2 md:grid-cols-4 lg:grid-cols-7 gap-3 mb-8">
          <StatCard label="Jumlah" value={stats.total} icon={<Book size={16}/>} color="bg-slate-100 text-slate-700" />
          <StatCard label="Tercapai" value={stats.tercapai} icon={<CheckCircle2 size={16}/>} color="bg-emerald-100 text-emerald-700" />
          <StatCard label="Gagal" value={stats.tidakTercapai} icon={<AlertCircle size={16}/>} color="bg-red-100 text-red-700" />
          <StatCard label="Exclusion" value={stats.exclusion} icon={<FilterX size={16}/>} color="bg-blue-100 text-blue-700" />
          <StatCard label="Denominator" value={stats.denominator} icon={<Activity size={16}/>} color="bg-indigo-100 text-indigo-700" />
          <StatCard label="Numerator" value={stats.numerator} icon={<BarChart3 size={16}/>} color="bg-teal-100 text-teal-700" />
          <StatCard label="Peratus KPI" value={`${stats.peratus}%`} icon={<PieChart size={16}/>} color="bg-amber-100 text-amber-700" isHighlight />
        </div>

        <div className="grid grid-cols-1 lg:grid-cols-4 gap-8">
          {/* Form Borang */}
          <div className="lg:col-span-1 print:hidden">
            <div className="bg-white p-6 rounded-2xl shadow-sm border border-slate-200 sticky top-8">
              <h2 className="text-xl font-semibold mb-6 flex items-center gap-2 text-slate-700 border-b border-slate-100 pb-3">
                {isEditing ? <Edit2 className="text-amber-500" size={20} /> : <Plus className="text-blue-600" size={20} />}
                {isEditing ? 'Kemaskini' : 'Rekod Baru'}
              </h2>
              
              <form onSubmit={handleSubmit} className="space-y-4">
                {showError && (
                  <div className="p-3 bg-red-50 border border-red-200 rounded-xl flex items-center gap-2 text-red-700 text-xs font-bold animate-pulse">
                    <AlertTriangle size={16} />
                    Waktu penyerahan tidak boleh mendahului waktu penerimaan!
                  </div>
                )}

                <div>
                  <label className="block text-xs font-bold mb-1.5 text-slate-600 uppercase tracking-tight">RN Buku Kematian</label>
                  <input 
                    required
                    name="rn_buku"
                    value={formData.rn_buku}
                    onChange={handleInputChange}
                    placeholder="Masukkan RN..."
                    className="w-full px-4 py-2 bg-slate-50 border border-slate-200 rounded-xl focus:ring-2 focus:ring-blue-500 outline-none transition-all font-semibold"
                  />
                </div>

                <div>
                  <label className="block text-xs font-bold mb-1.5 text-slate-600 uppercase tracking-tight">Exclusion Criteria</label>
                  <select 
                    name="exclusion"
                    value={formData.exclusion}
                    onChange={handleInputChange}
                    className="w-full px-4 py-2 bg-slate-50 border border-slate-200 rounded-xl focus:ring-2 focus:ring-blue-500 outline-none transition-all font-medium"
                  >
                    <option value="Tidak">Tidak</option>
                    <option value="Ya">Ya</option>
                  </select>
                </div>

                <div className={formData.exclusion === 'Ya' ? 'opacity-30 pointer-events-none' : ''}>
                  <label className={`block text-xs font-bold mb-1.5 uppercase tracking-tight ${showError ? 'text-red-600' : 'text-slate-600'}`}>
                    Masa Penerimaan
                  </label>
                  <input 
                    required={formData.exclusion === 'Tidak'}
                    type="datetime-local"
                    name="waktu_terima"
                    value={formData.waktu_terima}
                    onChange={handleInputChange}
                    className={`w-full px-4 py-2 bg-slate-50 border rounded-xl focus:ring-2 outline-none transition-all font-mono text-sm ${
                      showError ? 'border-red-300 ring-red-100' : 'border-slate-200 focus:ring-blue-500'
                    }`}
                  />
                </div>

                <div className={formData.exclusion === 'Ya' ? 'opacity-30 pointer-events-none' : ''}>
                  <label className={`block text-xs font-bold mb-1.5 uppercase tracking-tight ${showError ? 'text-red-600' : 'text-slate-600'}`}>
                    Masa Penyerahan
                  </label>
                  <input 
                    required={formData.exclusion === 'Tidak'}
                    type="datetime-local"
                    name="waktu_serah"
                    value={formData.waktu_serah}
                    onChange={handleInputChange}
                    className={`w-full px-4 py-2 bg-slate-50 border rounded-xl focus:ring-2 outline-none transition-all font-mono text-sm ${
                      showError ? 'border-red-300 ring-red-100' : 'border-slate-200 focus:ring-blue-500'
                    }`}
                  />
                </div>

                <div className="flex gap-2 pt-4">
                  <button 
                    type="submit"
                    disabled={showError}
                    className={`flex-1 flex items-center justify-center gap-2 py-3 rounded-xl font-bold text-white transition-all shadow-md active:scale-95 ${
                      showError ? 'bg-slate-300 cursor-not-allowed' : 
                      isEditing ? 'bg-amber-500 hover:bg-amber-600' : 'bg-blue-600 hover:bg-blue-700'
                    }`}
                  >
                    {isEditing ? <Save size={18} /> : <Plus size={18} />}
                    {isEditing ? 'Simpan' : 'Tambah'}
                  </button>
                  {isEditing && (
                    <button type="button" onClick={cancelEdit} className="px-4 py-3 bg-slate-200 text-slate-700 rounded-xl hover:bg-slate-300 transition-colors">
                      <X size={18} />
                    </button>
                  )}
                </div>
              </form>
            </div>
          </div>

          {/* Senarai Jadual */}
          <div className="lg:col-span-3 space-y-4">
            <div className="relative print:hidden">
              <Search className="absolute left-4 top-1/2 -translate-y-1/2 text-slate-400" size={20} />
              <input 
                type="text"
                placeholder="Cari RN..."
                value={searchTerm}
                onChange={(e) => setSearchTerm(e.target.value)}
                className="w-full pl-12 pr-4 py-3 bg-white border border-slate-200 rounded-2xl shadow-sm outline-none focus:ring-2 focus:ring-blue-500 transition-all"
              />
            </div>

            <div className="bg-white rounded-2xl shadow-sm border border-slate-200 overflow-hidden print:border-none">
              <table className="w-full text-left">
                <thead className="bg-slate-50 border-b border-slate-200 print:bg-white">
                  <tr>
                    <th className="px-4 py-3 text-xs font-bold text-slate-500 uppercase">RN & Waktu</th>
                    <th className="px-4 py-3 text-xs font-bold text-slate-500 uppercase text-center">Exclusion</th>
                    <th className="px-4 py-3 text-xs font-bold text-slate-500 uppercase text-center">Masa</th>
                    <th className="px-4 py-3 text-xs font-bold text-slate-500 uppercase text-center">KPI</th>
                    <th className="px-4 py-3 text-xs font-bold text-slate-500 uppercase text-right print:hidden">Tindakan</th>
                  </tr>
                </thead>
                <tbody className="divide-y divide-slate-100">
                  {filteredRecords.length > 0 ? (
                    filteredRecords.map((record) => (
                      <tr key={record.id} className="hover:bg-slate-50/50 transition-colors">
                        <td className="px-4 py-3">
                          <div className="font-bold text-slate-900">{record.rn_buku}</div>
                          {record.exclusion === 'Tidak' && record.waktu_terima && (
                            <div className="text-[10px] text-slate-400 font-medium">
                              T: {record.waktu_terima.replace('T', ' ')} | S: {record.waktu_serah.replace('T', ' ')}
                            </div>
                          )}
                        </td>
                        <td className="px-4 py-3 text-center text-xs font-bold">{record.exclusion}</td>
                        <td className="px-4 py-3 text-center text-sm font-mono font-bold">
                          {record.exclusion === 'Ya' ? '-' : `${record.jumlah_jam}j`}
                        </td>
                        <td className="px-4 py-3 text-center">
                          <span className={`inline-block px-2 py-0.5 rounded text-[10px] font-bold tracking-tight ${
                            record.status_kpi === 'Tercapai' ? 'bg-emerald-100 text-emerald-800' :
                            record.status_kpi === 'Exclusion' ? 'bg-blue-100 text-blue-800' : 'bg-red-100 text-red-800'
                          }`}>
                            {record.status_kpi}
                          </span>
                        </td>
                        <td className="px-4 py-3 text-right print:hidden">
                          <div className="flex justify-end gap-1">
                            <button onClick={() => startEdit(record)} className="p-1.5 text-amber-600 bg-amber-50 rounded-lg hover:bg-amber-100 transition-colors"><Edit2 size={14}/></button>
                            <button onClick={() => deleteRecord(record.id)} className="p-1.5 text-red-600 bg-red-50 rounded-lg hover:bg-red-100 transition-colors"><Trash2 size={14}/></button>
                          </div>
                        </td>
                      </tr>
                    ))
                  ) : (
                    <tr>
                      <td colSpan="5" className="px-6 py-12 text-center text-slate-400 italic font-medium">
                        Tiada rekod dijumpai bagi bulan ini.
                      </td>
                    </tr>
                  )}
                </tbody>
              </table>
            </div>
          </div>
        </div>
      </div>
      
      {/* Footer Print */}
      <div className="hidden print:block fixed bottom-0 left-0 right-0 text-center text-[10px] text-slate-500 py-4 border-t border-slate-200 bg-white italic font-bold">
        DATA PENDAFTARAN KEMATIAN - HOSPITAL TENGKU PERMAISURI NORASHIKIN | Dijana pada {new Date().toLocaleString('ms-MY')}
      </div>
    </div>
  );
};

// Sub-komponen untuk Kad Statistik
const StatCard = ({ label, value, icon, color, isHighlight }) => (
  <div className={`p-3 rounded-xl border border-slate-200 shadow-sm flex flex-col items-center justify-center text-center transition-transform hover:scale-105 ${isHighlight ? 'ring-2 ring-amber-400 bg-white' : 'bg-white'}`}>
    <div className={`p-1.5 rounded-lg mb-1.5 ${color}`}>
      {icon}
    </div>
    <span className="text-[10px] font-bold text-slate-400 uppercase tracking-tight leading-none mb-1">{label}</span>
    <span className={`text-sm font-bold tracking-tight ${isHighlight ? 'text-amber-600 text-lg' : 'text-slate-800'}`}>{value}</span>
  </div>
);

export default App;
